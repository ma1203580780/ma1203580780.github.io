function monkey_king($monkey,$piont){
    $i = 1;
    while(count($monkey) > 1){
        if($i%$piont == 0) {
            echo "干掉了 ".$monkey[$i-1].PHP_EOL;
            unset($monkey[$i-1]);
        }else{
          array_push($monkey, $monkey[$i-1]);
           unset($monkey[$i-1]);  
        }
        $i++;
    }
    return reset($monkey);    
}    
$king = monkey_king($monkey,$piont);
echo "KING is " .$king;

<?php
 /**
  * 简单洗牌算法
  */

 $card_num=54; //牌数
 print_r(wash_card($card_num));

 function wash_card($card_num)
 {
     $cards=$tmp=array();
     for($i=0;$i<$card_num;$i++){
         $tmp[$i]=$i;
     }

     for($i=0;$i<$card_num;$i++){
         $index=rand(0,$card_num-$i-1);
         $cards[$i]=$tmp[$index];
         unset($tmp[$index]);
         $tmp=array_values($tmp);
     }
     return $cards;
 }
